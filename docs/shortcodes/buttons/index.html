    Example #

    % 
clear
data = [1,2,3;4,5,6;7,8,9]
size_data = size(data);
data_expand = zeros(size_data(1)+2,size_data(2)+2);
data_expand(2:size_data(2)+1,2:size_data(2)+1) =data;
data_expand(2:size_data(2)+1,1) = data(:,2);
data_expand(2:size_data(2)+1,size_data(2)+2) = data(:,size_data(2)-1);
data_expand(1,:) = data_expand(3,:);
data_expand(size_data(1)+2,:) = data_expand(size_data(1),:);
% a_expand = 
mean_kernel =ones(3,3)
mean_kernel_sum = 9;
Gaussian_kernel = [1;2;1]*[1,2,1];
Gaussian_kernel_sum = 16;
% mean filter
y = imfilter(data_expand,mean_kernel);
mean_data_filter = y(2:size_data(1)+1,2:size_data(2)+1)
% gaussian filter
y = imfilter(data_expand,Gaussian_kernel);
Gaussian_data_filter = y(2:size_data(1)+1,2:size_data(2)+1);
